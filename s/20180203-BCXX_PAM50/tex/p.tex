m = Sequential([
	BatchNormalization(input_shape=(50,)),
	
	Dense(8*num_classes, kr=L2(1e-2)),
	
	Activation('softplus'),
	
	Dense(4*num_classes, kr=L2(1e-1)),
	
	Activation('softplus'),
	
	Dense(num_classes),
	
	Activation('softmax')
