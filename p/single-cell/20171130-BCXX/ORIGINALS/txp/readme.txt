	listEnsembl("GRCh=37")

	ensembl = useEnsembl(biomart="ensembl", dataset="hsapiens_gene_ensembl")
	listFilters(ensembl)
	head(listFilters(ensembl))
	genes <- getBM(attributes=c('ensembl_gene_id', 'hgnc_symbol','chromosome_name','start_position','end_position'), filters = 'ensembl_gene_id', values = RCCTPM$gene_id, mart = ensembl)
	genes <- genes[!genes$hgnc_symbol=="",]

	genes <- genes[match(unique(genes$hgnc_symbol),genes$hgnc_symbol), ]
